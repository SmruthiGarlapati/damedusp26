4. BST Insert & Search

Node* insert(Node* n, int val) {
    if (!n) return new Node(val);
    if (val < n->val) n->left  = insert(n->left,  val);
    else              n->right = insert(n->right, val);
    return n;
}

bool search(Node* n, int val) {
    if (!n) return false;
    if (n->val == val) return true;
    return val < n->val ? search(n->left, val)
                        : search(n->right, val);
}

5. Height & Node Count

int height(Node* n) {
    if (!n) return 0;
    return 1 + max(height(n->left), height(n->right));
}

int countNodes(Node* n) {
    if (!n) return 0;
    return 1 + countNodes(n->left) + countNodes(n->right);
}

Remember: A BST degrades to O(n) for a sorted input sequence because it becomes a linked list. This is why AVL and Red-Black trees add balancing.
